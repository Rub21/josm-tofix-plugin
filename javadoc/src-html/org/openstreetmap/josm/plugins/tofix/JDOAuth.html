228    private void getTofixAccessToken() {
229        try {
230            JsonObject json = OAuthClient.authTofixBackend(sessionId, requestToken);
231            if (json != null) {
232                String token = json.getString("tfx_token");
233                jTextArea1.setText(token);
234                setTofixToken(token);
235            }
236        } catch (Exception ex) {
237            JOptionPane.showMessageDialog(this, ex.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
238        }
239    }
240
241    public void autenticationOSM() {
242        try {
243            OAuthClient c = new OAuthClient();
244            sessionId = c.authorise(jTextField1.getText(), jPasswordField1.getText(), OAuthClient.getPrivileges());
245
246            requestToken = c.getRequestToken();
247            c.getOAuthToken(sessionId, requestToken);
248            Logging.info(sessionId.getToken());
249            setOsmOAuthToken(sessionId.getToken());
250            c.logout(sessionId);
251            jTabbedPane1.setSelectedIndex(1);
252            getTofixAccessToken();
253        } catch (Exception ex) {
254            JOptionPane.showMessageDialog(this, "Incorrect credentials : \n" + ex.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
255        }
256    }
