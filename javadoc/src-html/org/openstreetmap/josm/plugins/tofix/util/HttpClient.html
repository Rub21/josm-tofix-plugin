078    public Response connect(ProgressMonitor progressMonitor) throws IOException {
079        if (progressMonitor == null) {
080            progressMonitor = NullProgressMonitor.INSTANCE;
081        }
082        final HttpURLConnection connection = (HttpURLConnection) url.openConnection();
083        this.connection = connection;
084        connection.setRequestMethod(requestMethod);
085        connection.setRequestProperty("User-Agent", Version.getInstance().getFullAgentString());
086        connection.setConnectTimeout(connectTimeout);
087        connection.setReadTimeout(readTimeout);
088        connection.setInstanceFollowRedirects(false); // we do that ourselves
089        if (ifModifiedSince > 0) {
090            connection.setIfModifiedSince(ifModifiedSince);
091        }
092        connection.setUseCaches(useCache);
093        if (!useCache) {
094            connection.setRequestProperty("Cache-Control", "no-cache");
095        }
096        for (Map.Entry<String, String> header : headers.entrySet()) {
097            if (header.getValue() != null) {
098                connection.setRequestProperty(header.getKey(), header.getValue());
099            }
100        }
101
102        progressMonitor.beginTask(tr("Contacting Server..."), 1);
103        progressMonitor.indeterminateSubTask(null);
104
105        if ("PUT".equals(requestMethod) || "POST".equals(requestMethod) || "DELETE".equals(requestMethod)) {
106            Logging.info("{0} {1} ({2}) ...", requestMethod, url, Utils.getSizeString(requestBody.length, Locale.getDefault()));
107            if (Logging.isTraceEnabled() && requestBody.length > 0) {
108                Logging.trace("BODY: {0}", new String(requestBody, StandardCharsets.UTF_8));
109            }
110            connection.setFixedLengthStreamingMode(requestBody.length);
111            connection.setDoOutput(true);
112            try (OutputStream out = new BufferedOutputStream(
113                    new ProgressOutputStream(connection.getOutputStream(), requestBody.length,
114                            progressMonitor, outputMessage, finishOnCloseOutput))) {
115                out.write(requestBody);
116            }
117        }
118
119        boolean successfulConnection = false;
120        try {
121            try {
122                connection.connect();
123                final boolean hasReason = reasonForRequest != null && !reasonForRequest.isEmpty();
124                Logging.info("{0} {1}{2} -> {3}{4}",
125                        requestMethod, url, hasReason ? (" (" + reasonForRequest + ')') : "",
126                        connection.getResponseCode(),
127                        connection.getContentLengthLong() > 0
128                                ? (" (" + Utils.getSizeString(connection.getContentLengthLong(), Locale.getDefault()) + ')')
129                                : ""
130                );
131                if (Logging.isDebugEnabled()) {
132                    Logging.debug("RESPONSE: {0}", connection.getHeaderFields());
133                }
134                if (DefaultAuthenticator.getInstance().isEnabled() && connection.getResponseCode() == HttpURLConnection.HTTP_UNAUTHORIZED) {
135                    DefaultAuthenticator.getInstance().addFailedCredentialHost(url.getHost());
136                }
137            } catch (IOException | IllegalArgumentException | NoSuchElementException e) {
138                Logging.info("{0} {1} -> !!!", requestMethod, url);
139                Logging.warn(e);
140                //noinspection ThrowableResultOfMethodCallIgnored
141                Main.addNetworkError(url, Utils.getRootCause(e));
142                throw e;
143            }
144            if (isRedirect(connection.getResponseCode())) {
145                final String redirectLocation = connection.getHeaderField("Location");
146                if (redirectLocation == null) {
147                    /* I18n: argument is HTTP response code */
148                    throw new IOException(tr("Unexpected response from HTTP server. Got {0} response without ''Location'' header." +
149                            " Can''t redirect. Aborting.", connection.getResponseCode()));
150                } else if (maxRedirects > 0) {
151                    url = new URL(url, redirectLocation);
152                    maxRedirects--;
153                    Logging.info(tr("Download redirected to ''{0}''", redirectLocation));
154                    return connect();
155                } else if (maxRedirects == 0) {
156                    String msg = tr("Too many redirects to the download URL detected. Aborting.");
157                    throw new IOException(msg);
158                }
159            }
160            response = new Response(connection, progressMonitor);
161            successfulConnection = true;
162            return response;
163        } finally {
164            if (!successfulConnection) {
165                connection.disconnect();
166            }
167        }
168    }

174    public static final class Response {
175        private final HttpURLConnection connection;
176        private final ProgressMonitor monitor;
177        private final int responseCode;
178        private final String responseMessage;
179        private boolean uncompress;
180        private boolean uncompressAccordingToContentDisposition;
181        private String responseData;
182
183        private Response(HttpURLConnection connection, ProgressMonitor monitor) throws IOException {
184            CheckParameterUtil.ensureParameterNotNull(connection, "connection");
185            CheckParameterUtil.ensureParameterNotNull(monitor, "monitor");
186            this.connection = connection;
187            this.monitor = monitor;
188            this.responseCode = connection.getResponseCode();
189            this.responseMessage = connection.getResponseMessage();
190            if (this.responseCode >= 300) {
191                String contentType = getContentType();
192                if (contentType == null || (
193                        contentType.contains("text") ||
194                        contentType.contains("html") ||
195                        contentType.contains("xml"))
196                        ) {
197                    String content = this.fetchContent();
198                    if (content.isEmpty()) {
199                        Logging.debug("Server did not return any body");
200                    } else {
201                        Logging.debug("Response body: ");
202                        Logging.debug(this.fetchContent());
203                    }
204                } else {
205                    Logging.debug("Server returned content: {0} of length: {1}. Not printing.", contentType, this.getContentLength());
206                }
207            }
208        }

228        @SuppressWarnings("resource")
229        public InputStream getContent() throws IOException {
230            InputStream in;
231            try {
232                in = connection.getInputStream();
233            } catch (IOException ioe) {
234                Logging.debug(ioe);
235                in = Optional.ofNullable(connection.getErrorStream()).orElseGet(() -> new ByteArrayInputStream(new byte[]{}));
236            }
237            in = new ProgressInputStream(in, getContentLength(), monitor);
238            in = "gzip".equalsIgnoreCase(getContentEncoding()) ? new GZIPInputStream(in) : in;
239            Compression compression = Compression.NONE;
240            if (uncompress) {
241                final String contentType = getContentType();
242                Logging.debug("Uncompressing input stream according to Content-Type header: {0}", contentType);
243                compression = Compression.forContentType(contentType);
244            }
245            if (uncompressAccordingToContentDisposition && Compression.NONE.equals(compression)) {
246                final String contentDisposition = getHeaderField("Content-Disposition");
247                final Matcher matcher = Pattern.compile("filename=\"([^\"]+)\"").matcher(
248                        contentDisposition != null ? contentDisposition : "");
249                if (matcher.find()) {
250                    Logging.debug("Uncompressing input stream according to Content-Disposition header: {0}", contentDisposition);
251                    compression = Compression.byExtension(matcher.group(1));
252                }
253            }
254            in = compression.getUncompressedInputStream(in);
255            return in;
256        }

258        public BufferedReader getContentReader() throws IOException {
259            return new BufferedReader(
260                    UTFInputStreamReader.create(getContent())
261            );
262        }

264        public synchronized String fetchContent() throws IOException {
265            if (responseData == null) {
266                try (Scanner scanner = new Scanner(getContentReader()).useDelimiter("\\A")) { // \A - beginning of input
267                    responseData = scanner.hasNext() ? scanner.next() : "";
268                }
269            }
270            return responseData;
271        }

305        public Map<String, List<String>> getHeaderFields() {
306            Map<String, List<String>> ret = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
307            for (Entry<String, List<String>> e: connection.getHeaderFields().entrySet()) {
308                if (e.getKey() != null) {
309                    ret.put(e.getKey(), e.getValue());
310                }
311            }
312            return Collections.unmodifiableMap(ret);
313        }

403    private static boolean isRedirect(final int statusCode) {
404        switch (statusCode) {
405            case HttpURLConnection.HTTP_MOVED_PERM:
406            case HttpURLConnection.HTTP_MOVED_TEMP:
407            case HttpURLConnection.HTTP_SEE_OTHER:
408            case 307: // TEMPORARY_REDIRECT:
409            case 308: // PERMANENT_REDIRECT:
410                return true;
411            default:
412                return false;
413        }
414    }

420    private static void disconnect(final HttpURLConnection connection) {
421        if (connection != null) {
422            connection.setConnectTimeout(100);
423            connection.setReadTimeout(100);
424            try {
425                Thread.sleep(100);
426            } catch (InterruptedException ex) {
427                Logging.warn("InterruptedException in " + HttpClient.class + " during cancel");
428                Thread.currentThread().interrupt();
429            }
430            connection.disconnect();
431        }
432    }
