001package org.openstreetmap.josm.plugins.tofix.oauth.signpost;
002
003import org.openstreetmap.josm.plugins.tofix.oauth.signpost.http.HttpRequest;
004import org.openstreetmap.josm.plugins.tofix.oauth.signpost.http.HttpResponse;
005
006/**
007 * Provides hooks into the token request handling procedure executed by
008 * {@link OAuthProvider}.
009 * 
010 * @author Matthias Kaeppler
011 */
012public interface OAuthProviderListener {
013
014    /**
015     * Called after the request has been created and default headers added, but
016     * before the request has been signed.
017     * 
018     * @param request
019     *        the request to be sent
020     * @throws Exception
021     */
022    void prepareRequest(HttpRequest request) throws Exception;
023
024    /**
025     * Called after the request has been signed, but before it's being sent.
026     * 
027     * @param request
028     *        the request to be sent
029     * @throws Exception
030     */
031    void prepareSubmission(HttpRequest request) throws Exception;
032
033    /**
034     * Called when the server response has been received. You can implement this
035     * to manually handle the response data.
036     * 
037     * @param request
038     *        the request that was sent
039     * @param response
040     *        the response that was received
041     * @return returning true means you have handled the response, and the
042     *         provider will return immediately. Return false to let the event
043     *         propagate and let the provider execute its default response
044     *         handling.
045     * @throws Exception
046     */
047    boolean onResponseReceived(HttpRequest request, HttpResponse response) throws Exception;
048}
